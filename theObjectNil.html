@implementation DummyToCompileTheProgram : NSObject

-(int) sumOf:(int)value1 and:(int)value2 {
    return value1 + value2;
}

-(id) stringHello {
    return @"Hello";
}
@end

int main(int argc, const char * argv[])
{
    @autoreleasepool {
        id objVar = nil;
        int intVar = [objVar sumOf:10 and:5];

        NSLog(@"intVar = %i", intVar);
        NSLog(@"objectVar = %@", idVar);
    }
    
    return 0;
}
