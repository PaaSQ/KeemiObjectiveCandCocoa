#import <Foundation/Foundation.h>

// Person
@interface Person : NSObject

@property NSString* firstName;
@property (weak) NSString* lastName;
@property (unsafe_unretained) NSString* nickName;

@end

//
// main
//
int main(int argc, const char * argv[])
{
    // This code assumes Automatic Reference Counting ( ARC ) is used!
    @autoreleasepool {
        
        // We dont use constant strings for the first name, last name and nickname itself,
        // as constants are not relevant to ARC.
        const NSString * const fullName = @"Paul Johnson paul";
        
        Person *person = [Person new];
        @autoreleasepool {
            
            NSArray *names = [fullName componentsSeparatedByString:@" "];
            NSString *fn = names[0];
            NSString *ln = names[1];
            NSString *nn = names[2];
            person.firstName = fn;
            person.lastName = ln;
            person.nickName = nn;
            
        }
        
        if(person.firstName) {
            NSLog(@"firstName=%@", person.firstName);
        }
        
        if(person.lastName) {
            NSLog(@"lastName=%@", person.lastName);
        }
        
        if(person.nickName) {
            NSLog(@"nickName=%@", person.nickName);
        }
    }
    
    return 0;
}

// Person
@implementation Person


@end
