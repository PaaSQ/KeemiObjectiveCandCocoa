#import <Foundation/Foundation.h>

typedef enum {
    MY_ERROR_INVALID_PARAMETERS,
    MY_ERROR_CONNECTION_CANCELD,
    MY_ERROR_CONNECTION_LOSED,
} MyErrorCode;

typedef NS_ENUM(short, MyOption) {
    MY_OPTION_1,
    MY_OPTION_2,
    MY_OPTION_3,
    MY_OPTION_4,
};

typedef NS_OPTIONS(unsigned char, MyState) {
    MY_STATE_1 = 1 << 0,
    MY_STATE_2 = 1 << 1,
    MY_STATE_3 = 1 << 2,
    MY_STATE_4 = 1 << 3,
};

int main(int argc, const char * argv[])
{
    MyErrorCode error = MY_ERROR_INVALID_PARAMETERS;
    NSLog(@"length of error:%lu", sizeof(error));
    // prints: error:4
    
    MyOption option = MY_OPTION_1;
    NSLog(@"length of option:%lu", sizeof(option));
    // prints: option:2
    
    MyState state = MY_STATE_1;
    NSLog(@"length of state:%lu", sizeof(state));
    // prints: state:1
    
    return 0;
}
