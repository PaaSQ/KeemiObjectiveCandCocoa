#import <Foundation/Foundation.h>

typedef NS_ENUM(short, MyOption) {
    MY_OPTION_1,
    MY_OPTION_2,
    MY_OPTION_3,
    MY_OPTION_4,
};

int main(int argc, const char * argv[])
{
    
    MyOption option = MY_OPTION_1;
    switch (option) {
        case MY_OPTION_1:
            NSLog(@"option:%d", option);
            break;
    }
    
    return 0;
}
