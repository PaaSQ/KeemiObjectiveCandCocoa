- (void)dealloc {
    // CoreFoundation objects need to be explicitly released
    CFRelease(coreFoundationObject);
    // unregister for notifications so that they are not attempted to
    // be sent to a deallocated object
    [[NSNotificationCenter defaultCenter] removeObserver:self];
    // !!! DO NOT !!!
    // ... not free resources like file descriptors, sockets, or large blocks of memory
}
