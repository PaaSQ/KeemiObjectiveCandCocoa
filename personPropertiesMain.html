#import <Foundation/Foundation.h>
#import "Person.h"

int main(int argc, const char * argv[])
{
    @autoreleasepool {
        Person *person = [Person new];
        // Dot syntax allows you to access properties like this:
        person.firstName = @"Paul"; // equivalent to [person setFirstName:@"Paul"];
        person.lastName = @"Johnson"; // equivalent to [person lastName];
        NSLog(@"fullName=%@ %@", person.firstName, person.lastName);
    }
    return 0;
}
