#import <Foundation/Foundation.h>

// Person
@interface Person : NSObject

@property (nonatomic, copy) NSString* firstName;
@property (nonatomic, copy) NSString* lastName;

@end

//
// main
//
int main(int argc, const char * argv[])
{
    // This code assumes Automatic Reference Counting ( ARC ) is used!
    @autoreleasepool {
        
        // We dont use constant strings for the first name, last name and nickname itself,
        // as constants are not relevant to ARC.
        const NSString * const fullName = @"Paul Johnson";
        
        NSArray *names = [fullName componentsSeparatedByString:@" "];
        NSMutableString *fnm = [NSMutableString stringWithString:names[0]];
        NSMutableString *lnm = [NSMutableString stringWithString:names[1]];
        Person *person = [Person new];
        person.firstName = fnm;
        person.lastName = lnm;
        [fnm setString:@"1234"];
        [lnm setString:@"1234567"];
        
        if(person.firstName) {
            NSLog(@"firstName=%@", person.firstName);
        }
        
        if(person.lastName) {
            NSLog(@"lastName=%@", person.lastName);
        }
    }
    
    return 0;
}

// Person
@implementation Person

-(void)setLastName:(NSString *)lastName {
    _lastName = lastName;
}

@end
